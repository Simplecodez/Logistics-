CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    user_id UUID NOT NULL,
    full_name VARCHAR(255) NOT NULL,
    email VARCHAR(200) UNIQUE NOT NULL,
    phone_number VARCHAR(50) UNIQUE NOT NULL,
	user_role users_role DEFAULT 'user',
    password VARCHAR(200) NOT NULL,
    is_active BOOLEAN DEFAULT FALSE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    activation_token VARCHAR(200),
	activation_token_time bigint
);



CREATE TABLE packages (
    id SERIAL PRIMARY KEY,
    package_id UUID NOT NULL,
    package_name VARCHAR(255) NOT NULL,
    receiver_email VARCHAR(200) NOT NULL,
    receiver_phone_number VARCHAR(50) NOT NULL,
	user_ref UUID REFERENCES users(user_id) NOT NULL,
    tracking_password VARCHAR(200) NOT NULL,
    status package_status DEFAULT 'Registered/Not received',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
	status_time_stamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
	delivery_fee smallint DEFAULT 2000,
	pick_up_date TIMESTAMP NOT NULL,
	estimated_time_of_arrival TIMESTAMP NOT NULL,
	pick_up_location GEOMETRY(Point, 4326) NOT NULL,
	exact_location GEOMETRY(Point, 4326) NOT NULL	
);
